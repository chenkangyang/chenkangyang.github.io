使用c++实现一个词法分析器

@Kangyang Chen · May 4, 2019 · 6 min read

c++实现一个c语言的词法分析器，对单词进行识别和编码

要求

                   +-------+                      +--------+
-- source code --> | lexer | --> token stream --> | parser | --> assembly
                   +-------+                      +--------+

设计并实现一个词法分析器，实现对指定位置的类C语言源程序文本文件的读取，并能够对该源程序中的所有单词进行分类，指出其所属类型，实现简单的词法分析操作。

  • 允许用户自己输入源程序并保存为文件
  • 系统能够输出经过预处理后的源程序（去掉注释、换行、空格等）
  • 能够将该源程序中所有的单词根据其所属类型（整数、保留字、运算符、标识符等。定义的类C语言中的标识符只能以字母或下划线开头）进行归类显示，例如：识别保留字：if、int、for、while、do、return、break、continue等，其他的都识别为标识符；常数为无符号整形数；运算符包括：+、-、*、/、=、>、<、>=、<=、!=等；分隔符包括：,、;、{、}、(、)等。
  • 实现文件的读取操作，而不是将文本以字符串形式预存于程序中。文本内容为待分析的类C语言程序。

例如下面为一段C语言源程序：

main()
{
int  a,b;
a = 10;
   b = a + 20;
}

要求输出如下:(可以自行分类)

（1，’main’）
（5，’（’）
（5，’）’）
（5，’{ ’）
（1，’int’）
（2，’a’）
（5，’,’）
（2，’b’）
（5，’;’）
（2，’a’）
（4，’=’）
（3，’10’）
（5，’;’）
（2，’b’）
（4，’=’）
（2，’a’）
（4，’+’）
（3，’20’）
（5，’;’）
（5，’}’）

设计

  1. 完成文件读取

  2. 文本识别 将源代码识别为5类符号：保留字、标识符、整常数、运算符、界符 定义5类符号的集合为：

    1：保留字(33) auto break case char const continue
    default do double else enum extern
    float for goto if int long
    register return short signed sizeof static
    struct switch typedef union unsigned void
    volatile while define

    2：标识符 letter(letter | digit)* 无穷集

    3：常数(无符号整形数) (digit)+ 无穷集

    4：运算符 <、<=、>、>=、=、+、-、*、/、^、等

    5：界符 #、(、)、{、}、;、[、]、,、;、、”、’ 等

    "[", "]", ",", ";", '"', '\''}

完整程序

#include <iostream>
#include <cstring>
#include <fstream>

using namespace std;

#define Max 500  //最大代码长度

//词的结构，二元组形式（单词种别，单词自身的值）
typedef struct
{
    int typenum; //单词种别
    char *word;
}Table;

//全局变量, 保留字表
static char reserveWordTbl[34][20] = {
        "auto", "break", "case", "char", "const", "continue",
        "default", "do", "double", "else", "enum", "extern",
        "float", "for", "goto", "if", "int", "long",
        "register", "return", "short", "signed", "sizeof", "static",
        "struct", "switch", "typedef", "union", "unsigned", "void",
        "volatile", "while", "printf", "define"
};

//运算符表, 根据需要可以自行增加
static char operatorTbl[10][4]={
    "<", "<=", ">", ">=", "=", "+", "-", "*", "/", "^"
};

// 界符
static char delimiterTbl[12][3]={
        "#", "(", ")", "{", "}", ";",
        "[", "]", ",", ";", '"', '\''
};

/***************** 过滤源代码 ************************/

void Skip_comments(char *r) {
    char tempString[Max];  //源代码中需要被保留部分
    int count = 0; //用于指向tempString

    // 跳过注释，无用字符
    for (int i = 0; i <= strlen(r); i++)
    {
        if (r[i] == '/'&&r[i + 1] == '/')
        {//若为单行注释“//”,则去除注释后面的东西，直至遇到回车换行
            while (r[i] != '\n')
            {
                i++;//向后扫描，直到遇到回车，也就跳过了注释
            }
        }
        if (r[i] == '/'&&r[i + 1] == '*')
        {//若为多行注释“/* 。。。*/”则去除该内容
            i += 2;
            while (r[i] != '*' || r[i + 1] != '/')
            {
                i++;//继续扫描
                if (r[i] == '$')
                {
                    printf("注释出错，没有找到 */，程序结束！！！\n");
                    exit(0);
                }
            }
            i += 2;//跨过“*/”
        }
        tempString[count++] = r[i];
    }

    tempString[count] = '\0';
    strcpy(r, tempString);//产生净化之后的源程序
}
/***************** 过滤源代码 ************************/


/***************** 输入源代码、空的结构体数组指针、 ************************/

int Scanner(char *r, Table *table) {
    int chIndex = 0;
    int wordIndex = 0;
    cout << "源程序长度: " << strlen(r) << endl;
    // 去除注释
    Skip_comments(r);
    cout << "去除注释后的源程序长度: " << strlen(r) << endl;
    // cout << "去除注释后的程序: " << endl << r << endl;
    // 取出保留字

    // 遍历各字符
    while (chIndex < strlen(r)) {
        // cout << "index: "<< chIndex << "char: "<< r[chIndex] << endl;

        while (r[chIndex] == ' ' || r[chIndex] == '\n' || r[chIndex] == '\t' || r[chIndex] == '\v' ||
               r[chIndex] == '\r') {
            // cout << "空格之类的, index: " << chIndex << endl;
            if (chIndex < strlen(r)) {
                chIndex++;
            }
        }

        // 以字母开头, 可能是标识符, 也可能是保留字
        if (isalpha(r[chIndex])) {
            char tmp[256];
            int tmpIndex = 0;
            tmp[tmpIndex++] = r[chIndex++];
            while (isdigit(r[chIndex]) || isalpha(r[chIndex])) {
                tmp[tmpIndex++] = r[chIndex++];
            }
            tmp[tmpIndex] = '\0';
            // cout << "得到: " << tmp << endl;
            bool isReserve = false;
            // 判断是否是保留字
            for (int i = 0; i < 34; i++) {
                if (strcmp(tmp, reserveWordTbl[i]) == 0) {
                    table[wordIndex].typenum = 1;
                    table[wordIndex].word = new char[20];
                    strcpy(table[wordIndex].word, tmp);
                    // cout << "保留字: " << table[wordIndex].word << endl;
                    wordIndex++;
                    isReserve = true;
                    break;
                }
            }
            if (!isReserve) {
                table[wordIndex].word = new char[20];
                strcpy(table[wordIndex].word, tmp);
                table[wordIndex].typenum = 2;
                // cout << "标识符: " << table[wordIndex].word << endl;
                wordIndex++;
            }
        } else if (isdigit(r[chIndex])) {   // 遇到数字
            char tmp[20];
            int tmpIndex = 0;
            while (isdigit(r[chIndex])) {
                tmp[tmpIndex++] = r[chIndex++];
            }
            tmp[tmpIndex] = '\0';
            table[wordIndex].word = new char[20];
            strcpy(table[wordIndex].word, tmp);
            table[wordIndex].typenum = 3;
            // cout << "数字: " << tmp << endl;
        } else if (r[chIndex] == '<' || r[chIndex] == '>' || r[chIndex] == '=' || r[chIndex] == '+' ||
                   r[chIndex] == '-' || r[chIndex] == '*' || r[chIndex] == '/' || r[chIndex] == '^') {
            char tmp[5];
            int tmpIndex = 0;
            if (r[chIndex] == '<' && r[chIndex + 1] == '=') {
                tmp[tmpIndex++] = r[chIndex++];
            } else if (r[chIndex] == '>' && r[chIndex + 1] == '=') {
                tmp[tmpIndex++] = r[chIndex++];
            } else {
                tmp[tmpIndex++] = r[chIndex++];
            }
            tmp[tmpIndex] = '\0';
            for (int i = 0; i < 10; i++) {
                if (strcmp(tmp, operatorTbl[i]) == 0) {
                    table[wordIndex].word = new char[5];
                    strcpy(table[wordIndex].word, tmp);
                    table[wordIndex].typenum = 4;
                    // cout << "运算符" << table[wordIndex].word << endl;
                    wordIndex++;
                    break;
                }
            }
        } else if (r[chIndex] == '#' || r[chIndex] == '(' || r[chIndex] == ')' || r[chIndex] == '{' ||
                   r[chIndex] == '}' || r[chIndex] == '[' || r[chIndex] == ']' || r[chIndex] == ',' ||
                   r[chIndex] == ';' || r[chIndex] == '"' || r[chIndex] == '\'') {
            char tmp[3];
            int tmpIndex = 0;
            tmp[tmpIndex++] = r[chIndex++];
            tmp[tmpIndex] = '\0';
            // cout << "得到界符: " << tmp << endl;
            for (int i = 0; i < 11; i++) {
                if (strcmp(tmp, delimiterTbl[i]) == 0) {
                    table[wordIndex].word = new char[20];
                    strcpy(table[wordIndex].word, tmp);
                    table[wordIndex].typenum = 5;
                    // cout << "界符: " << table[wordIndex].word << endl;
                    wordIndex++;
                    break;
                }
            }
        } else {
            cout << "在index="<<chIndex<<"处遇到陌生字符: " << r[chIndex] << "，跳过" <<endl;
            chIndex ++;
        }
    }
    return wordIndex;
}


int main() {

    ifstream in;
    ofstream out;
    char inputFileName[50];
    char outputFileName[50];
    char resource[Max]; // 存放输入代码的缓冲区
    Table *table = new Table[Max]; // 存放二元组
    int wordNum = 0;
    int choice;
    cout << "请输入读入方式：1 从文件读， 2 从键盘读（结束标志位$)）" << endl;

    cin >> choice;
    switch (choice){
        // 从文件读取
        case 1:
            cout << "输入读入文件的名称"<<endl;
            cin >> inputFileName;
            in.open(inputFileName, ifstream::in);
            if (in.fail()) {
                cout << "打开文件失败." << endl;
                exit(1);
            }
            in.getline(resource, Max, '$'); // 读入文件内容至缓冲区resource中
            cout << "源程序:" << endl << resource << endl;
            wordNum = Scanner(resource, table);
            cout << "扫描完毕，识别出" << wordNum << "个字符" << endl;
            for (int i = 0; i < wordNum; i++) {
                cout << "(" << table[i].typenum << ", " << table[i].word << ")" << endl;
            }
            break;
            // 从键盘读入，已'$'结尾
        case 2:
            cin.getline(resource, Max,'$');
            cout << "输入保存文件的名称"<<endl;
            cin >> outputFileName;
            out.open(outputFileName, ofstream::out);
            if (out.fail()) {
                cout << "打开文件失败." << endl;
                exit(1);
            }
            out << resource;
            wordNum = Scanner(resource, table);
            cout << "扫描完毕，识别出" << wordNum << "个字符" << endl;
            for (int i = 0; i < wordNum; i++) {
                cout << "(" << table[i].typenum << ", " << table[i].word << ")" << endl;
            }
            break;
    }

    return 0;
}
